NCF learns a probabilistic model that emphasizes the binary property of implicit data. NCF explores the use of DNNs for collaborative filtering, by using a multi-layer perceptron (MLP) to learn the user-item interaction function. Lastly, here we discussed a new neural matrix factorization model called NeuMF, which ensembles MF and MLP under the NCF framework; it unifies the strengths of linearity of MF and non-linearity of MLP for modeling the user-item latent structures. The last segment contains a working example of NCF

def generate_id_mappings(ids_list):

    inverse_Id_map = {old_id : new_id for new_id, old_id in enumerate(ids_list)}
    return (Id_map,inverse_Id_map)

def make_pivot(data):
    df = pd.DataFrame(data,  columns=['u', 'i', 'r'])
    return pd.pivot_table(df, index='u', columns='i', values='r')

def negative_sample(pivot_table, size=100, max_try=10):
    try_        = 0
    sample_size = 0
    samples     = []
    remaining   = size
    index_size  = len(pivot_table.index)
    columns_size = len(pivot_table.columns)
    while sample_size < size and try_ < max_try:
        ind_index   = np.random.choice(index_size, remaining)
        ind_columns = np.random.choice(columns_size, remaining)
        for i in range(remaining):
            x, y = ind_index[i], ind_columns[i]
            if pivot_table.iloc[x, y] == 1: pass
            samples.append((pivot_table.index[x], pivot_table.columns[y], 0))
            sample_size += 1
        remaining = size - sample_size 
        try_ += 1
        
    return np.array(samples)

def train(model, x_train, optimizer, criterion, epochs, lr, fold=40,x_test=None, verbose=False, evaluate_training=False):
    history_train = []
    history_test  = []
    optimizer     = optimizer(model.parameters(), lr=lr)
     
    ind   = 0
    data  = np.array_split(x_train, fold)
    
    model.train()
    for i in tqdm(range(epochs)):
        # take a batch the create negative sample
        positive_sampling = data[ind % fold]
        negative_sampling = negative_sample(make_pivot(positive_sampling), size=len(positive_sampling))
        
        samples = np.vstack([positive_sampling, negative_sampling])
        
        np.random.shuffle(samples)
        
        users_list, items_list, labels = samples[:, 0], samples[:, 1], samples[:, 2] 
        users   = torch.LongTensor(users_list)
        items   = torch.LongTensor(items_list)
        ratings = torch.FloatTensor(labels)
        # comupute the loss 
        comp_rui = model(users, items)
        loss     = criterion(comp_rui, ratings)
        #Save the current loss
        history_train.append(loss.item())
        
        # Backpropagation
        loss.backward()
        optimizer.step()
        
        ind += 1

        if verbose and (i% SHOW_EVERY == 0): print("loss %.3f" % (loss.item()))

        if evaluate_training:
            assert x_test is not None
            history_test += [evaluate(model, x_test, criterion, eval=False)]
        
    return history_train, history_test
                  

def evaluate(model, x_test, criterion, eval=True):
    if eval : model.eval()
    
    users_list, items_list, rating_list = x_test[:, 0], x_test[:, 1], x_test[:, 2]

    users   = torch.LongTensor(users_list)
    items   = torch.LongTensor(items_list)
    ratings = torch.FloatTensor(rating_list)

    comp_rui = model(users, items)
    loss = criterion(comp_rui, ratings)
    return loss.item()

class NeuMF(torch.nn.Module):
    def __init__(self, config=config, size_users=nb_users, size_items=nb_items, size_factors=nb_factors, dropout=False):
        super(NeuMF, self).__init__()
        
        self.nb_users = size_users
        self.nb_items = size_items
        self.factors  = size_factors
        self.dropout  = dropout
        assert config['dropout_rate'] < 1
        self.dropout_rate = config['dropout_rate']
        
        # mf part
        self.embedding_user_mf = torch.nn.Embedding(num_embeddings=self.nb_users, embedding_dim=self.factors)
        self.embedding_item_mf = torch.nn.Embedding(num_embeddings=self.nb_items, embedding_dim=self.factors)        
        
        # mlp part
        self.embedding_user_mlp = torch.nn.Embedding(num_embeddings=self.nb_users, embedding_dim=self.factors)
        self.embedding_item_mlp = torch.nn.Embedding(num_embeddings=self.nb_items, embedding_dim=self.factors)        
        
        
        self.fc_layers = torch.nn.ModuleList()
        for in_size, out_size in zip(config['layers'][:-1], config['layers'][1:]):
            self.fc_layers += [ torch.nn.Linear(in_size, out_size) ]
        
        self.fusion_layer  = torch.nn.Linear(in_features=config['layers'][-1] * 2, out_features=1)
        self.sigmoid = torch.nn.Sigmoid()
        
        
        
    def forward(self, user_indices, item_indices):
        user_embedding_mf  = self.embedding_user_mf(user_indices)
        item_embedding_mf  = self.embedding_item_mf(item_indices)
        
        user_embedding_mlp = self.embedding_user_mlp(user_indices)
        item_embedding_mlp = self.embedding_item_mlp(item_indices)
        
        
        # mf part
        mf_vector = torch.mul(user_embedding_mf, item_embedding_mf)
        
        if self.dropout : mf_vector = torch.nn.Dropout(p=self.dropout_rate)(mf_vector)
        
        # mlp part
        mlp_vector = torch.cat([user_embedding_mlp, item_embedding_mlp], dim=-1) # the concat latentvector
        
        for idx in range(len(self.fc_layers)):
            mlp_vector = self.fc_layers[idx](mlp_vector)
            mlp_vector = torch.nn.ReLU()(mlp_vector)

        if self.dropout : mlp_vector = torch.nn.Dropout(p=self.dropout_rate)(mlp_vector)
        
        vector = torch.cat([mlp_vector, mf_vector], dim=-1)
        fusion = self.fusion_layer(vector)
        output = self.sigmoid(fusion)
        return output.flatten()

    def set_GMF_emb(self, emb_users, emb_items):
        self.embedding_user_mf.weight.data = emb_users
        self.embedding_item_mf.weight.data = emb_items


    def set_MLP_emb(self, emb_users, emb_items):
        self.embedding_user_mlp.weight.data = emb_users
        self.embedding_item_mlp.weight.data = emb_items

class GMF(nn.Module):
    def __init__(self, num_users=nb_users, num_items=nb_items, nb_factors=nb_factors, generalized=True):
        super(GMF, self).__init__()
        self.generalized = generalized

        self.user_emb = nn.Embedding(num_users, nb_factors)
        self.item_emb = nn.Embedding(num_items, nb_factors)
        # The linear layer bias replace the user and item bias
        self.linear = torch.nn.Linear(nb_factors*2, 1)
        
        # if not generalized,use the model as a regressor not as a proba estimator
        if not self.generalized:
            self.linear.weight.data.fill_(1)
            self.user_emb.weight.data.normal_(0, .1)
            self.item_emb.weight.data.normal_(0, .1)

    def forward(self, u, i):
        user = self.user_emb(u)
        item = self.item_emb(i)
        x    = torch.cat([user, item], 1)
        out  = torch.sigmoid(self.linear(x))
        return out.flatten()

class MLP(nn.Module):

    def __init__(self, config=config, size_users=nb_users, size_items=nb_items, size_factors=nb_factors, dropout=False):
        super(MLP, self).__init__()
        
        self.nb_users = size_users
        self.nb_items = size_items
        self.factors  = size_factors
        self.dropout  = dropout
        assert config['dropout_rate'] < 1
        self.dropout_rate = config['dropout_rate']
        self.sigmoid = torch.nn.Sigmoid()
        
        self.embedding_users = torch.nn.Embedding(num_embeddings=self.nb_users, embedding_dim=self.factors)
        self.embedding_items = torch.nn.Embedding(num_embeddings=self.nb_items, embedding_dim=self.factors)        
        
        self.fc_layers = torch.nn.ModuleList()
        for in_size, out_size in zip(config['layers'][:-1], config['layers'][1:]):
            self.fc_layers += [ torch.nn.Linear(in_size, out_size) ]
        
        self.last_layer  = torch.nn.Linear(in_features=config['layers'][-1], out_features=1)

    def forward(self, user_indices, item_indices):
        
        user_embedding = self.embedding_users(user_indices)
        item_embedding = self.embedding_items(item_indices)
                
        vector = torch.cat([user_embedding, item_embedding], dim=-1) # concat latent vector
        
        for idx in range(len(self.fc_layers)):
            vector = self.fc_layers[idx](vector)
            vector = torch.nn.ReLU()(vector)
        
        if self.dropout :  vector = torch.nn.Dropout(p=self.dropout_rate)(vector)

        vector = self.last_layer(vector)
        output = self.sigmoid(vector)
        return output.flatten()

class RMSELoss(nn.Module):
    def __init__(self, eps=1e-6):
        super().__init__()
        self.mse = nn.MSELoss()
        self.eps = eps
        
    def forward(self,yhat,y):
        loss = torch.sqrt(self.mse(yhat,y) + self.eps)
        return loss

def hitRatio(model, test_data, threshold=.8):
    users, items = test_data[:, 0], test_data[:, 1]
    users = torch.LongTensor(users)
    items = torch.LongTensor(items)
    groundTruth = model(users, items)
    return (groundTruth >= threshold).sum() / len(groundTruth), groundTruth
